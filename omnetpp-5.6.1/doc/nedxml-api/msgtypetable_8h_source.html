99  bool iscOwnedObject; // field type is derived from cOwnedObject

103  std::string mutableReturnType; // mutableGetter C++ return type
104  std::string var; // name of data member variable
105  std::string argName; // setter argument name
106  std::string sizeVar; // data member to store size of dynamic array
107  std::string sizeType; // type of array sizes and array indices
108  std::string getter; // getter function name: "T getter() const;" "const T& getter() const" default value is getFoo
109  std::string getterForUpdate; // mutable getter function name: "T& getterForUpdate();" default value is getFooForUpdate
110  bool hasGetterForUpdate; // whether a getterForUpdate method needs to be generated
111  bool allowReplace; // @allowReplace; whether setter of an owned pointer field is allowed to delete an already-set object
112  std::string dropper; // dropper function name (for owned pointers: drop pointer from class and return to caller)
113  std::string clone; // @clone; code to duplicate (one array element of) the field (for owned pointers)

115  std::string inserter; // @inserter; inserter function name (insert element to dynamic array)
116  std::string eraser; // @eraser; eraser function name (erase element from dynamic array)
117  std::string sizeSetter; // setArraySize() function name
118  std::string sizeGetter; // array size getter function name
119  std::string toString; // function to convert data to string, defined in property @toString
120  // if tostring begins with a dot, then it is taken as member function call, parentheses needed in property; otherwise toString is understood as name of a normal function, do not use parentheses
121  // std::string <function>(<datatype>); // @toString(function)
122  // std::string <function>(const <datatype>&); // @toString(function)
123  // const char * <function>(<datatype>); // @toString(function)
124  // const char * <function>(const <datatype>&); // @toString(function)
125  // std::string <datatype>::<function>(...); // @toString(.function(...))
126  // const char * <datatype>::<function>(...); // @toString(.function(...))
127  std::string fromString; // function to convert string to data member, defined in property @fromString
128  // <datatype> <function>(const char *); // @fromString(function)
129  std::string getterConversion; // currently only with strings: ".c_str()"
130  std::string enumName; // from @enum
131  std::string enumQName; // fully qualified type name of enum
132  bool nopack; // @nopack(true)
133  bool isEditable; // @editable(true): field value is editable in the UI via the descriptor's setFieldValueFromString() method
134  bool isOpaque; // @opaque(true), means that field type is treated as atomic (has no fields), i.e. has no descriptor
135  bool overrideGetter; // @overrideGetter|@override, used when field getter function overrides a function in base class
136  bool overrideSetter; // @overrideSetter|@override, used when field setter function overrides a function in base class
